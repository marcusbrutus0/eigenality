**Goal:** Add an SEO/PageSpeed/accessibility audit system that runs during dev mode, serves results at `/_audit` endpoints, and shows a per-page overlay badge.

**Architecture:** A new `src/build/audit/` module runs as the final build phase, inspecting `SiteConfig` and rendered HTML in `dist/` via `lol_html`. Results are written as HTML, JSON, and Markdown files. In dev mode, a floating overlay badge is injected into each page. A standalone `eigen audit` CLI command exposes the same checks for CI/AI workflows.

**Tech Stack:** Rust, lol_html (HTML inspection), serde/serde_json (JSON output), existing SiteConfig/RenderedPage types.

**Spec:** `docs/superpowers/specs/2026-03-18-dev-audit-design.md`

```rust
//! in `dist/` to produce an `AuditReport` with actionable findings.

use crate::config::SiteConfig;
use eyre::Result;
use serde::Serialize;
use std::collections::BTreeMap;
use std::path::Path;

/// Severity of an audit finding.
#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, Serialize)]
#[serde(rename_all = "lowercase")]
pub enum Severity {
    Critical,
    High,
    Medium,
    Low,
}

/// Category of an audit check.
#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, Serialize)]
#[serde(rename_all = "snake_case")]
pub enum Category {
    Seo,
    Performance,
    Accessibility,
    BestPractices,
}

impl Category {
    pub fn display_name(&self) -> &'static str {
        match self {
            Self::Seo => "SEO",
            Self::Performance => "Performance",
            Self::Accessibility => "Accessibility",
            Self::BestPractices => "Best Practices",
        }
    }
}

impl Severity {
    pub fn display_name(&self) -> &'static str {
        match self {
            Self::Critical => "CRITICAL",
            Self::High => "HIGH",
            Self::Medium => "MEDIUM",
            Self::Low => "LOW",
        }
    }
}

/// Scope of a check: site-wide or per-page.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize)]
#[serde(rename_all = "lowercase")]
pub enum Scope {
    Site,
    Page,
}

/// Actionable fix information for a finding.
#[derive(Debug, Clone, Serialize)]
pub struct Fix {
    /// File to modify (e.g., "site.toml" or "templates/about.html").
    pub file: String,
    /// Human-readable instruction for the fix.
    pub instruction: String,
}

/// A single audit finding.
#[derive(Debug, Clone, Serialize)]
pub struct Finding {
    /// Stable check ID (e.g., "seo/meta-description").
    pub id: &'static str,
    pub category: Category,
    pub severity: Severity,
    pub scope: Scope,
    /// Human-readable issue description.
    pub message: String,
    /// Actionable fix.
    pub fix: Fix,
}

/// Summary counts for the audit report.
#[derive(Debug, Clone, Serialize)]
pub struct AuditSummary {
    pub total: usize,
    pub by_severity: BTreeMap<Severity, usize>,
    pub by_category: BTreeMap<Category, usize>,
}

/// The complete audit report.
#[derive(Debug, Clone, Serialize)]
pub struct AuditReport {
    pub summary: AuditSummary,
    pub site_findings: Vec<Finding>,
    pub page_findings: BTreeMap<String, Vec<Finding>>,
}

impl AuditReport {
    /// Build summary counts from the findings.
    fn compute_summary(
        site_findings: &[Finding],
        page_findings: &BTreeMap<String, Vec<Finding>>,
    ) -> AuditSummary {
        let mut by_severity: BTreeMap<Severity, usize> = BTreeMap::new();
        let mut by_category: BTreeMap<Category, usize> = BTreeMap::new();
        let mut total = 0;

        let all_findings = site_findings
            .iter()
            .chain(page_findings.values().flatten());

        for f in all_findings {
            total += 1;
            *by_severity.entry(f.severity).or_default() += 1;
            *by_category.entry(f.category).or_default() += 1;
        }

        AuditSummary {
            total,
            by_severity,
            by_category,
        }
    }
}

/// Run the full audit against the built site.
///
/// Inspects config for site-level checks, then reads each rendered page's
/// HTML from `dist/` for page-level checks. Filters out any check IDs
/// listed in `config.audit.ignore`.
pub fn run_audit(
    config: &SiteConfig,
    dist_path: &Path,
    rendered_pages: &[RenderedPage],
) -> Result<AuditReport> {
    let ignore_ids: Vec<&str> = config
        .audit
        .as_ref()
        .map(|a| a.ignore.iter().map(|s| s.as_str()).collect())
        .unwrap_or_default();

    // Run site-level checks.
    let mut site_findings = checks::run_site_checks(config, dist_path);
    site_findings.retain(|f| !ignore_ids.contains(&f.id));

    // Run page-level checks.
    let mut page_findings: BTreeMap<String, Vec<Finding>> = BTreeMap::new();
    for page in rendered_pages {
        let html_path = dist_path.join(page.url_path.trim_start_matches('/'));
        let html = match std::fs::read_to_string(&html_path) {
            Ok(h) => h,
            Err(_) => continue,
        };

        let template_path = page
            .template_path
            .as_deref()
            .unwrap_or("(unknown)");

        let mut findings =
            checks::run_page_checks(&html, &page.url_path, template_path, config);
        findings.retain(|f| !ignore_ids.contains(&f.id));

        if !findings.is_empty() {
            page_findings.insert(page.url_path.clone(), findings);
        }
    }

    let summary = AuditReport::compute_summary(&site_findings, &page_findings);

    Ok(AuditReport {
        summary,
        site_findings,
        page_findings,
    })
}
```

```rust
use super::super::{Category, Finding, Fix, Scope, Severity};
use crate::config::SiteConfig;
use std::path::Path;

pub fn site_checks(config: &SiteConfig, dist_path: &Path) -> Vec<Finding> {
    let mut findings = Vec::new();

    // seo/sitemap — check if sitemap.xml was generated
    if !dist_path.join("sitemap.xml").exists() {
        findings.push(Finding {
            id: "seo/sitemap",
            category: Category::Seo,
            severity: Severity::Critical,
            scope: Scope::Site,
            message: "No sitemap.xml found. Search engines use sitemaps to discover pages.".into(),
            fix: Fix {
                file: "site.toml".into(),
                instruction: "Sitemap is generated automatically when pages are rendered. Verify the build completes successfully.".into(),
            },
        });
    }

    // seo/robots-txt — check if robots.txt is configured
    if config.robots.is_none() {
        findings.push(Finding {
            id: "seo/robots-txt",
            category: Category::Seo,
            severity: Severity::High,
            scope: Scope::Site,
            message: "No robots.txt configured. Search engine crawlers look for robots.txt to understand crawl rules.".into(),
            fix: Fix {
                file: "site.toml".into(),
                instruction: "Add a [robots] section:\n[robots]\nsitemap = true\n[[robots.rules]]\nuser_agent = \"*\"\nallow = [\"/\"]".into(),
            },
        });
    }

    // seo/feed — check if any feed is configured
    if config.feed.is_empty() {
        findings.push(Finding {
            id: "seo/feed",
            category: Category::Seo,
            severity: Severity::Low,
            scope: Scope::Site,
            message: "No Atom feed configured. Feeds help readers and aggregators discover new content.".into(),
            fix: Fix {
                file: "site.toml".into(),
                instruction: "Add a [feed.<name>] section with source, path, title, and author fields.".into(),
            },
        });
    }

    findings
}
```

```rust
    let in_head = Rc::new(RefCell::new(false));

    let ih = in_head.clone();
    let bs = blocking_scripts.clone();
    let ins = imgs_no_srcset.clone();
    let img_opt_enabled = config.assets.images.optimize;

    let _ = lol_html::rewrite_str(
        html,

            element_content_handlers: vec![
                lol_html::element!("head", move |_el| {
                    *ih.borrow_mut() = true;
                    Ok(())
                }),
                lol_html::element!("script[src]", {
                    let in_head = in_head.clone();
                    let bs = bs.clone();
                    move |el| {
                        if *in_head.borrow() {
                            let has_defer = el.get_attribute("defer").is_some();
                            let has_async = el.get_attribute("async").is_some();
                            if !has_defer && !has_async {
                                let src = el.get_attribute("src").unwrap_or_default();
                                bs.borrow_mut().push(src);
                            }
                        }
                        Ok(())
                    }
                }),
                lol_html::element!("img[src]", move |el| {
                    if !img_opt_enabled && el.get_attribute("srcset").is_none() {
                        let src = el.get_attribute("src").unwrap_or_default();
                        ins.borrow_mut().push(src);
                    }
                    Ok(())
                }),
            ],
            ..lol_html::RewriteStrSettings::new()
        },

    let mut results = Vec::new();
```

```rust
pub fn page_checks(html: &str, _page_path: &str, template_path: &str) -> Vec<Finding> {
    use std::cell::RefCell;
    use std::rc::Rc;

    let tpl = template_path.to_string();
    let has_lang = Rc::new(RefCell::new(false));
    let has_viewport = Rc::new(RefCell::new(false));
    let imgs_no_alt: Rc<RefCell<Vec<String>>> = Rc::new(RefCell::new(Vec::new()));
    // For link text: track <a> elements and their accumulated text content.
    // We use a simple approach: collect hrefs of anchors that have no text content.
    let empty_links: Rc<RefCell<Vec<String>>> = Rc::new(RefCell::new(Vec::new()));
    let current_link_href: Rc<RefCell<Option<String>>> = Rc::new(RefCell::new(None));
    let current_link_text: Rc<RefCell<String>> = Rc::new(RefCell::new(String::new()));

    let hl = has_lang.clone();
    let hv = has_viewport.clone();
    let ina = imgs_no_alt.clone();
    let clh = current_link_href.clone();
    let clt = current_link_text.clone();
    let el2 = empty_links.clone();

    let _ = lol_html::rewrite_str(
        html,
        lol_html::RewriteStrSettings {
            element_content_handlers: vec![
                lol_html::element!("html[lang]", move |_el| {
                    *hl.borrow_mut() = true;
                    Ok(())
                }),
                lol_html::element!("meta[name='viewport']", move |el| {
                    if let Some(content) = el.get_attribute("content") {
                        if content.contains("width=device-width") {
                            *hv.borrow_mut() = true;
                        }
                    }
                    Ok(())
                }),
                lol_html::element!("img", move |el| {
                    if el.get_attribute("alt").is_none() {
                        let src = el.get_attribute("src").unwrap_or_default();
                        ina.borrow_mut().push(src);
                    }
                    Ok(())
                }),
                // Track <a> elements: on open, store href and reset text buffer.
                lol_html::element!("a[href]", {
                    let clh = clh.clone();
                    let clt = clt.clone();
                    let el2 = el2.clone();
                    move |el| {
                        // If there was a previous unclosed link, check it.
                        if let Some(href) = clh.borrow_mut().take() {
                            let text = clt.borrow().trim().to_string();
                            if text.is_empty() {
                                el2.borrow_mut().push(href);
                            }
                        }
                        *clh.borrow_mut() = el.get_attribute("href");
                        clt.borrow_mut().clear();

                        // Set up end tag handler to check accumulated text.
                        let clh2 = current_link_href.clone();
                        let clt2 = current_link_text.clone();
                        let el3 = empty_links.clone();
                        el.on_end_tag(move |_end| {
                            if let Some(href) = clh2.borrow_mut().take() {
                                let text = clt2.borrow().trim().to_string();
                                if text.is_empty() {
                                    el3.borrow_mut().push(href);
                                }
                            }
                            Ok(())
                        })?;
                        Ok(())
                    }
                }),
            ],
            // Capture text inside <a> elements.
            document_content_handlers: vec![
                lol_html::text!("a", {
                    let clt = current_link_text.clone();
                    move |text| {
                        clt.borrow_mut().push_str(text.as_str());
                        Ok(())
                    }
                }),
            ],
            ..lol_html::RewriteStrSettings::new()
        },
    );

    let mut results = Vec::new();

    if !*has_lang.borrow() {
        results.push(Finding {
            id: "a11y/html-lang",
            category: Category::Accessibility, severity: Severity::Critical, scope: Scope::Page,
            message: "Missing lang attribute on <html>. Screen readers need this to select the correct pronunciation.".into(),
            fix: Fix { file: tpl.clone(), instruction: "Add lang=\"en\" (or appropriate language) to the <html> tag.".into() },
        });
    }

    if !*has_viewport.borrow() {
        results.push(Finding {
            id: "a11y/viewport-meta",
            category: Category::Accessibility, severity: Severity::Critical, scope: Scope::Page,
            message: "Missing or misconfigured <meta name=\"viewport\">. Must include width=device-width for mobile.".into(),
            fix: Fix {
                file: tpl.clone(),
                instruction: "Add <meta name=\"viewport\" content=\"width=device-width, initial-scale=1\"> to <head>.".into(),
            },
        });
```

```rust
use super::super::{AuditReport, Severity};
use eyre::Result;
use std::fmt::Write;
use std::path::Path;

pub fn render_markdown(report: &AuditReport) -> String {
    let mut md = String::new();
    let _ = writeln!(md, "# Eigen Audit Report\n");
    let _ = writeln!(md, "## Summary\n");

    for sev in &[Severity::Critical, Severity::High, Severity::Medium, Severity::Low] {
        let count = report.summary.by_severity.get(sev).copied().unwrap_or(0);
        if count > 0 {
            let _ = writeln!(md, "- **{}:** {}", sev.display_name(), count);
        }
    }

    if report.summary.total == 0 {
        let _ = writeln!(md, "\nAll checks passed!");
        return md;
    }

    if !report.site_findings.is_empty() {
        let _ = writeln!(md, "\n## Site-Wide Issues\n");
        for f in &report.site_findings {
            let _ = writeln!(md, "### [{}] {}", f.severity.display_name(), f.id);
            let _ = writeln!(md, "{}", f.message);
            let _ = writeln!(md, "**Fix:** In `{}`, {}\n", f.fix.file, f.fix.instruction);
        }
    }

    if !report.page_findings.is_empty() {
        let _ = writeln!(md, "\n## Page Issues\n");
        for (page, findings) in &report.page_findings {
            let _ = writeln!(md, "### {}\n", page);
            for f in findings {
                let _ = writeln!(md, "#### [{}] {}", f.severity.display_name(), f.id);
                let _ = writeln!(md, "{}", f.message);
                let _ = writeln!(md, "**Fix:** In `{}`, {}\n", f.fix.file, f.fix.instruction);
            }
        }
    }

    md
}

pub fn write_markdown(report: &AuditReport, dist_path: &Path) -> Result<()> {
    let md = render_markdown(report);
    std::fs::write(dist_path.join("_audit.md"), md)?;
    Ok(())
}
```

```rust
/// Discover rendered pages by walking dist/ for HTML files.
/// Used by `eigen audit --no-build` when the build pipeline's
/// rendered_pages list is not available.
fn discover_rendered_pages(dist_dir: &Path) -> Result<Vec<build::render::RenderedPage>> {
    use std::path::Path;
    let mut pages = Vec::new();
    for entry in walkdir::WalkDir::new(dist_dir)
        .into_iter()
        .filter_map(|e| e.ok())
        .filter(|e| e.path().extension().map(|x| x == "html").unwrap_or(false))
    {
        let rel = entry.path().strip_prefix(dist_dir)?;
        let rel_str = rel.to_string_lossy();
        // Skip audit files, fragments, and error pages.
        if rel_str.starts_with("_audit")
            || rel_str.starts_with("_fragments")
            || rel_str.starts_with("_error")
        {
            continue;
        }
        let url_path = format!("/{}", rel_str);
        let is_index = rel_str.ends_with("index.html");
        pages.push(build::render::RenderedPage {
            url_path,
            is_index,
            is_dynamic: false,
            template_path: None,
        });
    }
    Ok(pages)
}
```
